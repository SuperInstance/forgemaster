__global__ void dcs_simulation(float* results, int trials, float noise_start, float noise_step, int noise_steps) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = gridDim.x * blockDim.x;

    for (int n = 0; n < noise_steps; n++) {
        float p_noise = noise_start + n * noise_step;
        int success_count = 0;

        // Each thread runs multiple trials
        for (int t = 0; t < trials; t++) {
            // Simulate DCS: assume deterministic correct signal is 1.0, noise is random in [-p_noise, p_noise]
            float noise = (2.0f * (float)rand() / RAND_MAX - 1.0f) * p_noise;
            float signal = 1.0f + noise;

            // DCS "breaks" when noise flips the sign (simple model of zero noise tolerance)
            if (signal > 0.0f) {
                success_count++;
            }
        }

        if (idx == 0) {
            float success_rate = (float)success_count / trials;
            results[n] = success_rate;
        }
    }
}

int main() {

    const int noise_steps = 20;
    const float noise_start = 0.0f;

    // Analyze threshold: find where success rate drops below 0.5
    float threshold = 0.0f;
    int threshold_idx = -1;
    for (int i = 0; i < noise_steps; i++) {
        printf("Noise=%.2f, SuccessRate=%.3f\n", noise_start + i * noise_step, h_results[i]);
        if (threshold_idx == -1 && h_results[i] < 0.5f) {
            threshold_idx = i;

        }
    }

    // Determine if transition is sharp or gradual
    int sharp = 1;
    for (int i = 0; i < noise_steps - 1; i++) {
        float diff = fabsf(h_results[i] - h_results[i+1]);
        if (diff < 0.3f && h_results[i] > 0.8f && h_results[i+1] < 0.2f) {
            // Large drop in one step
        } else if (diff < 0.3f) {
            sharp = 0; // gradual
        }
    }

    if (threshold_idx == -1) threshold = noise_start + noise_steps * noise_step;

    printf("SUMMARY: DCS breaks at noise threshold %.2f; transition is %s\n",

    free(h_results);
