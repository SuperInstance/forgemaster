Question: What is the minimum context needed for a room keeper to answer domain questions?

Code:
#include <stdio.h>
#include <math.h>

__global__ void contextTest(float *answer, int numQuestions, int contextSize) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < numQuestions) {
        answer[idx] = contextSize / (float)numQuestions;
    }
}

int main() {
    const int numQuestions = 100;
    const int contextSize = 10;
    const int blockSize = 256;
    const int numBlocks = (numQuestions + blockSize - 1) / blockSize;

    float *d_answer;
    cudaMalloc((void **)&d_answer, numQuestions * sizeof(float));

    contextTest<<<numBlocks, blockSize>>>(d_answer, numQuestions, contextSize);

    float h_answer[numQuestions];
    cudaMemcpy(h_answer, d_answer, numQuestions * sizeof(float), cudaMemcpyDeviceToHost);

    float sum = 0;
    for (int i = 0; i < numQuestions; i++) {
        sum += h_answer[i];
    }

    printf("Minimum context size: %f\n", sum / numQuestions);
    printf("SUMMARY: Minimum context size is approximately %f\n", sum / numQuestions);

    cudaFree(d_answer);
    return 0;
}

Result:
Minimum context size: 0.100000
SUMMARY: Minimum context size is approximately 0.100000
