Question: How does CT snap interact with gradient descent? Does snapping weights help or hurt training?

Code:
#include <cuda_runtime.h>
#include <cmath>

#define NUM_ITERATIONS 100
#define LEARNING_RATE 0.01
#define WEIGHT_SNAP 0.5

__global__ void train(float *weights, float *gradient, bool snap) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < 1000) {
        float weight = weights[idx];
        float grad = gradient[idx];
        for (int i = 0; i < NUM_ITERATIONS; i++) {
            weight -= LEARNING_RATE * grad;
            if (snap && weight > WEIGHT_SNAP) weight = WEIGHT_SNAP;
            weights[idx] = weight;
        }
    }
}

int main() {
    float *weights, *gradient;
    cudaMalloc((void **)&weights, 1000 * sizeof(float));
    cudaMalloc((void **)&gradient, 1000 * sizeof(float));

    float h_weights[1000], h_gradient[1000];
    for (int i = 0; i < 1000; i++) {
        h_weights[i] = 1.0f;
        h_gradient[i] = -0.1f;
    }
    cudaMemcpy(weights, h_weights, 1000 * sizeof(float), cudaMemcpyHostToDevice);
    cudaMemcpy(gradient, h_gradient, 1000 * sizeof(float), cudaMemcpyHostToDevice);

    train<<<10, 100>>>(weights, gradient, false);
    cudaDeviceSynchronize();
    float sum_no_snap;
    cudaMemcpy(&sum_no_snap, weights, sizeof(float), cudaMemcpyDeviceToHost);
    for (int i = 1; i < 1000; i++) {
        float temp;
        cudaMemcpy(&temp, weights + i, sizeof(float), cudaMemcpyDeviceToHost);
        sum_no_snap += temp;
    }

    cudaMemset(weights, 0, 1000 * sizeof(float));
    cudaMemcpy(weights, h_weights, 1000 * sizeof(float), cudaMemcpyHostToDevice);
    train<<<10, 100>>>(weights, gradient, true);
    cudaDeviceSynchronize();
    float sum_snap;
    cudaMemcpy(&sum_snap, weights, sizeof(float), cudaMemcpyDeviceToHost);
    for (int i = 1; i < 1000; i++) {
        float temp;
        cudaMemcpy(&temp, weights + i, sizeof(float), cudaMemcpyDeviceToHost);
        sum_snap += temp;
    }

    printf("Average weight without snap: %f\n", sum_no_snap / 1000.0f);
    printf("Average weight with snap: %f\n", sum_snap / 1000.0f);
    printf("SUMMARY: Snapping weights hurts training if average weight without snap (%f) is closer to optimal weight (0) than with snap (%f).\n", sum_no_snap / 1000.0f, sum_snap / 1000.0f);

    cudaFree(weights);
    cudaFree(gradient);
    return 0;
}
