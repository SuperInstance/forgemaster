Question: What is the entropy of a CT-snapped signal vs raw float? Information-theoretic comparison.

Code:
#include <stdio.h>
#include <math.h>

#define N 1048576

__global__ void entropy_kernel(float *data, float *ct_snapped, float *entropy_raw, float *entropy_ct) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= N) return;

    float raw = data[idx];
    float ct = roundf(raw * 1000.0f) / 1000.0f; // CT snapping to 3 decimal places

    ct_snapped[idx] = ct;

    if (isnan(raw)) raw = 0.0f;
    if (isnan(ct)) ct = 0.0f;

    entropy_raw[idx] = -raw * log2f(fabsf(raw));
    entropy_ct[idx] = -ct * log2f(fabsf(ct));
}

int main() {
    float *data, *ct_snapped, *entropy_raw, *entropy_ct;
    cudaMallocManaged(&data, N * sizeof(float));
    cudaMallocManaged(&ct_snapped, N * sizeof(float));
    cudaMallocManaged(&entropy_raw, N * sizeof(float));
    cudaMallocManaged(&entropy_ct, N * sizeof(float));

    // Initialize data with random values
    for (int i = 0; i < N; i++) {
        data[i] = (float) rand() / RAND_MAX;
    }

    int blockSize = 256;
    int numBlocks = (N + blockSize - 1) / blockSize;
    entropy_kernel<<<numBlocks, blockSize>>>(data, ct_snapped, entropy_raw, entropy_ct);
    cudaDeviceSynchronize();

    float sum_entropy_raw = 0.0f;
    float sum_entropy_ct = 0.0f;
    for (int i = 0; i < N; i++) {
        sum_entropy_raw += entropy_raw[i];
        sum_entropy_ct += entropy_ct[i];
    }

    printf("Entropy of raw float signal: %f\n", sum_entropy_raw / N);
    printf("Entropy of CT-snapped signal: %f\n", sum_entropy_ct / N);
    printf("SUMMARY: Entropy of CT-snapped signal (%f) vs raw float signal (%f)\n", sum_entropy_ct / N, sum_entropy_raw / N);

    cudaFree(data);
    cudaFree(ct_snapped);
    cudaFree(entropy_raw);
    cudaFree(entropy_ct);

    return 0;
}

Result:
Entropy of raw float signal: 0.360980
Entropy of CT-snapped signal: nan
SUMMARY: Entropy of CT-snapped signal (nan) vs raw float signal (0.360980)
