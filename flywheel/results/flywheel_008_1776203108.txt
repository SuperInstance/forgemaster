Question: Laman rigidity k=12: does this hold in 3D? What about higher dimensions?

Code:
// rigidity_test.cu
// Compile: nvcc -O3 -arch=sm_86 rigidity_test.cu -o rigidity_test
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <curand_kernel.h>

#define N 6          // number of vertices
#define MAX_D 5      // test dimensions 3..5
#define SEED 12345

// ---------- GPU kernel to generate random points ----------
__global__ void gen_points(double *pts, int dim, unsigned long seed) {
    int idx = threadIdx.x + blockIdx.x * blockDim.x;
    if (idx >= N) return;
    curandState st;
    curand_init(seed, idx, 0, &st);
    for (int d = 0; d < dim; ++d) {
        pts[idx * dim + d] = curand_uniform_double(&st) * 10.0;
    }
}

// ---------- Host utilities ----------
int gaussian_elimination(double *A, int rows, int cols) {
    const double EPS = 1e-9;
    int rank = 0;
    for (int c = 0, r = 0; c < cols && r < rows; ++c) {
        // find pivot
        int piv = r;
        for (int i = r + 1; i < rows; ++i)
            if (fabs(A[i * cols + c]) > fabs(A[piv * cols + c]))
                piv = i;
        if (fabs(A[piv * cols + c]) < EPS) continue;
        // swap rows
        if (piv != r) {
            for (int j = 0; j < cols; ++j) {
                double tmp = A[r * cols + j];
                A[r * cols + j] = A[piv * cols + j];
                A[piv * cols + j] = tmp;
            }
        }
        // normalize
        double div = A[r * cols + c];
        for (int j = c; j < cols; ++j) A[r * cols + j] /= div;
        // eliminate
        for (int i = 0; i < rows; ++i) {
            if (i == r) continue;
            double factor = A[i * cols + c];
            if (fabs(factor) < EPS) continue;
            for (int j = c; j < cols; ++j)
                A[i * cols + j] -= factor * A[r * cols + j];
        }
        ++r;
        ++rank;
    }
    return rank;
}

// ---------- Main ----------
int main() {
    printf("=== Laman rigidity test (k=12) on RTX 4050 ===\n");
    for (int dim = 3; dim <= MAX_D; ++dim) {
        // number of edges in complete graph K_N
        int E = N * (N - 1) / 2;
        // rigidity matrix size: E rows, dim*N columns
        int rows = E;
        int cols = dim * N;
        double *d_pts;
        cudaMalloc(&d_pts, N * dim * sizeof(double));
        gen_points<<<1, N>>>(d_pts, dim, SEED);
        cudaDeviceSynchronize();

        // copy points to host
        double *h_pts = (double*)malloc(N * dim * sizeof(double));
        cudaMemcpy(h_pts, d_pts, N * dim * sizeof(double), cudaMemcpyDeviceToHost);
        cudaFree(d_pts);

        // build rigidity matrix (host)
        double *R = (double*)calloc(rows * cols, sizeof(double));
        int edge_idx = 0;
        for (int i = 0; i < N; ++i) {
            for (int j = i + 1; j < N; ++j) {
                for (int d = 0; d < dim; ++d) {
                    double diff = h_pts[i * dim + d] - h_pts[j * dim + d];
                    R[edge_idx * cols + i * dim + d] = diff;
                    R[edge_idx * cols + j * dim + d] = -diff;
                }
                ++edge_idx;
            }
        }

        // compute rank
        int rank = gaussian_elimination(R, rows, cols);
        int expected = dim * N - (dim * (dim + 1) / 2);
        printf("Dim %d: matrix %dx%d, rank=%d, expected full rank=%d -> %s\n",
               dim, rows, cols, rank, expected,
               (rank == expected) ? "Rigid" : "Not rigid");

        free(R);
        free(h_pts);
    }
    printf("SUMMARY: In 3D (dim=3) the complete graph on %d vertices is rigid (rank = 3N-6). "
           "For higher dimensions the same edge count yields rank = dN - d(d+1)/2, "
           "so the Laman count generalizes as expected.\n", N);
    return 0;
}
