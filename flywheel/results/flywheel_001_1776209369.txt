Question: Does CT snap preserve topology? Can connected components split after snapping?

Code:
#include <stdio.h>
#include <math.h>

#define NUM_POINTS 1000
#define NUM_ITERATIONS 10

__global__ void ct_snap(float* points, int num_points, float* snapped_points) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= num_points) return;

    float x = points[idx * 2];
    float y = points[idx * 2 + 1];

    // Simple snapping function: round to nearest integer
    snapped_points[idx * 2] = round(x);
    snapped_points[idx * 2 + 1] = round(y);
}

__global__ void check_topology(float* points, int num_points, float* snapped_points, int* result) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= num_points) return;

    float x = points[idx * 2];
    float y = points[idx * 2 + 1];
    float snapped_x = snapped_points[idx * 2];
    float snapped_y = snapped_points[idx * 2 + 1];

    // Check if connected components split
    for (int i = 0; i < num_points; i++) {
        if (idx!= i && hypot(x - points[i * 2], y - points[i * 2 + 1]) < 1.0f) {
            if (hypot(snapped_x - snapped_points[i * 2], snapped_y - snapped_points[i * 2 + 1]) > 1.0f) {
                *result = 1;
                return;
            }
        }
    }
}

int main() {
    float* points;
    float* snapped_points;
    int* result;

    cudaMallocManaged(&points, NUM_POINTS * 2 * sizeof(float));
    cudaMallocManaged(&snapped_points, NUM_POINTS * 2 * sizeof(float));
    cudaMallocManaged(&result, sizeof(int));

    // Initialize points
    for (int i = 0; i < NUM_POINTS; i++) {
        points[i * 2] = (float)rand() / RAND_MAX;
        points[i * 2 + 1] = (float)rand() / RAND_MAX;
    }

    *result = 0;

    ct_snap<<<NUM_POINTS / 256, 256>>>(points, NUM_POINTS, snapped_points);
    check_topology<<<NUM_POINTS / 256, 256>>>(points, NUM_POINTS, snapped_points, result);

    cudaDeviceSynchronize();

    printf("Connected components split: %d\n", *result);
    SUMMARY: printf("SUMMARY: CT snap does %s preserve topology\n", *result == 1? "not" : "");

    cudaFree(points);
    cudaFree(snapped_points);
    cudaFree(result);

    return 0;
}
