### 1. **Does Golden-Ratio Hashing of 5D→2D Cut-and-Project Coordinates Exponentially Suppress Collision Clustering Compared to Linear Congruent Hashing?**

**Python Test (<100 lines):**

```python
    pts_5d = np.random.randint(0, 12, size=(n, 5))
    # Orthogonal Penrose 5D→2D projection
    P = np.array([[np.cos(2*np.pi*k/5), -np.sin(2*np.pi*k/5)] for k in range(5)]).T
    return pts_5d @ P % 1.0

def golden_hash(pts, grid=200):
    phi = 1.618033988749895
    # Irrational multiplier prevents rational lattice aliasing
    u = pts @ np.array([phi, phi**2])
    return np.floor((u % 1.0) * grid).astype(int)

def linear_hash(pts, grid=200):
    return np.floor((pts[:,0]*0.618 + pts[:,1]*0.382) * grid).astype(int) % grid

def collision_rate(pts, n, h_fn, grid=200):
    h = h_fn(pts, grid)
    return 1.0 - len(np.unique(h)) / len(h)

pts = generate_5d_2d(5000)
gc = collision_rate(pts, 5000, golden_hash)
lc = collision_rate(pts, 5000, linear_hash)
print(f"Golden: {gc:.4f}, Linear: {lc:.4f}")
# FALSIFIABLE if gc >= lc (no advantage)
```

### 2. **Do Fibonacci-Word-Modulated Projection Windows Induce Detectable Spectral Anisotropy in the 2D Memory Index?**
**Why it's falsifiable:** The hypothesis claims that using Fibonacci words to scale the cut-and-project window creates non-random directional bias. If the spectral coherence of Fibonacci-modulated coordinates is statistically indistinguishable from random window scaling, the claim of structured anisotropy is falsified.

**Python Test (<100 lines):**
```python
import numpy as np

def fib_windows(k):
    f = [0, 1]
    while len(f) < k: f.append(f[-1] + f[-2])
    return np.array(f[-k:]) % 0.9 + 0.1  # Scale to (0.1, 1.0)

def project_with_window(pts_2d, w):
    return pts_2d * w % 1.0  # Simulate window thickness modulation

def spectral_bias(pts_2d, bins=32):
    hist, _, _ = np.histogram2d(pts_2d[:,0], pts_2d[:,1], bins=bins)
    fft_vals = np.abs(np.fft.rfft2(hist))**2
    peak = fft_vals.max()
    baseline = np.percentile(fft_vals[1:], 50)
    return peak / (baseline + 1e-9)

n, k = 3000, 12
pts = np.random.rand(n, 2)
fib_ws = fib_windows(k)
cohs_fib = [spectral_bias(project_with_window(pts, w)) for w in fib_ws]

# Control: uniformly random window factors
cohs_rand = [spectral_bias(project_with_window(pts, np.random.uniform(0.1, 1.0))) for _ in range(k)]
print(f"Fib bias mean: {np.mean(cohs_fib):.3f}, Rand bias mean: {np.mean(cohs_rand):.3f}")
# FALSIFIABLE if mean(fib) ≤ mean(rand)
```

```python
    idx = np.argmin(np.abs(dists - target_r))
    err = abs(dists[idx] - target_r)
    _ = time.perf_counter() - t0
    return err
```
